description: "Get Typing Club Bot running in under two minutes — no installation required."

<Warning>
Do not change the `minDelay` or `maxDelay` values from `60`. If the delay is too low, TypingClub's result page will not appear and the bot will stall.
</Warning>

<Steps>
  <Step title="Open TypingClub and navigate to a lesson">
    Go to [typingclub.com](https://www.typingclub.com), log in, and open the first typing lesson you want to automate. Make sure the lesson is fully loaded and the typing interface is visible before continuing.

    <Info>
      The bot starts from whichever lesson is currently displayed. To automate from a specific point, navigate directly to that lesson first.
    </Info>
  </Step>

  <Step title="Open the browser developer console">
    <Tabs>
      <Tab title="Chrome">
        Press `Ctrl` + `Shift` + `J` on Windows/Linux, or `Cmd` + `Option` + `J` on macOS.

        Alternatively: right-click anywhere on the page → **Inspect** → click the **Console** tab.

        <Info>
          Chrome may show a warning: "Don't paste code you don't understand or trust." You can safely dismiss this — the script only interacts with the TypingClub page you have open.
        </Info>
      </Tab>
      <Tab title="Firefox">
        Press `Ctrl` + `Shift` + `K` on Windows/Linux, or `Cmd` + `Option` + `K` on macOS.

        Alternatively: right-click anywhere on the page → **Inspect** → click the **Console** tab.
      </Tab>
      <Tab title="Safari">
        First enable the Develop menu: **Safari** → **Settings** → **Advanced** → check **Show features for web developers**.

        Then press `Cmd` + `Option` + `C`, or use **Develop** → **Show JavaScript Console**.
      </Tab>
      <Tab title="Edge">
        Press `Ctrl` + `Shift` + `J` on Windows, or `Cmd` + `Option` + `J` on macOS.
      </Tab>
    </Tabs>
  </Step>

  <Step title="Copy the full script">
    Copy the entire script below:

    ```javascript TypingClubBot.js
    /**
     * This script types for you automatically on www.typingclub.com:
     * 1. Open the website
     * 2. Blaze past the tutorials
     * 3. Go into a level
     * 4. Open Console
     * 5. Paste the script and press ENTER
     */

    // NOTE: When delay (in ms between two strokes) is too low, the site might bug out and the result page will not be shown
    const minDelay = 60;
    const maxDelay = 60;

    const keyOverrides = {
      [String.fromCharCode(160)]: ' '    // convert hardspace to normal space
    };

    function getTargetCharacters() {
      const els = Array.from(document.querySelectorAll('.token span.token_unit'));
      const chrs = els
        .map(el => {
          // get letter to type from each letter DOM element
          if (el.firstChild?.classList?.contains('_enter')) {
            // special case: ENTER
            return '\n';
          }
          let text = el.textContent[0];
          return text;
        })
        .map(c => keyOverrides.hasOwnProperty(c) ? keyOverrides[c] : c); // convert special characters
      return chrs;
    }

    function recordKey(chr) {
      // send it straight to the internal API
      window.core.record_keydown_time(chr);
    }

    function sleep(ms) {
      return new Promise(r => setTimeout(r, ms));
    }

    async function autoPlay() {
      const chrs = getTargetCharacters();
      if (chrs.length === 0) {
        // No characters to type, move to the next level
        goToNextLevel();
        return;
      }

      for (let i = 0; i < chrs.length; ++i) {
        const c = chrs[i];
        recordKey(c);
        await sleep(Math.random() * (maxDelay - minDelay) + minDelay);
      }

      // After auto-typing ends, automatically move to the next level
      goToNextLevel();
    }

    // Function to click the button that moves to the next level
    function goToNextLevel() {
      // Wait a bit after typing before attempting to go to the next level
      setTimeout(async () => {
        const nextButton = Array.from(document.querySelectorAll('span')).find(el => el.textContent === '→');
        if (nextButton) {
          nextButton.click(); // Simulate click to proceed to the next level
          console.log('Moving to next level...');

          // Wait for the next level to load before starting the typing again
          await sleep(1000); // Adjust this delay if necessary
          autoPlay(); // Call autoPlay again for the next level
        } else {
          console.log('Next level button not found.');
        }
      }, 2000); // Adjust the delay before going to the next level if needed
    }

    autoPlay();
    ```
  </Step>

  <Step title="Paste into the console and press Enter">
    Click inside the console input area, paste the copied script, and press **Enter**.

    The bot will immediately begin reading and typing the current lesson's characters.
  </Step>

  <Step title="Watch the bot run">
    The bot will:
    1. Type all characters in the current lesson at ~60 ms per keystroke
    2. Wait 2 seconds after finishing, then click the **→** next-level button
    3. Wait 1 second for the next lesson to load
    4. Begin typing the next lesson automatically
    5. Repeat until all levels are complete or the next-level button is not found

    <Tip>
      Keep your laptop plugged in and the TypingClub browser tab active and visible. The tab must remain in the foreground for keystrokes to register correctly. Do not minimize the window or switch to another tab.
    </Tip>

    If the console prints `Next level button not found.`, the bot has reached a non-typing lesson (such as a video). Manually advance to the next typing lesson and paste the script again.
  </Step>
</Steps>
